3. Model (Deep MLP)¶

We build a deep multi-layer perceptron (MLP) for FashionMNIST classification.

Notes:

  • Input images are (1, 28, 28) so we flatten to a vector of length 784.
  • Output layer has 10 logits (one per class).
  • We do not apply softmax in the model because CrossEntropyLoss expects raw logits.

import torch.nn as nn

class DeepMLP(nn.Module):
    def __init__(self):
        super().__init__()

        # Sequential stack
        self.net = nn.Sequential(
            nn.Flatten(),                 # (B, 1, 28, 28) -> (B, 784)

            nn.Linear(28*28, 512),
            nn.ReLU(),

            nn.Linear(512, 256),
            nn.ReLU(),

            nn.Linear(256, 128),
            nn.ReLU(),

            nn.Linear(128, 10)            # logits for 10 classes
        )

    def forward(self, x):
        return self.net(x)

4.1 Accuracy function¶

This evaluates accuracy on a dataset/list by creating a DataLoader internally. We use model.eval() and torch.no_grad() so evaluation is deterministic and efficient.

In [11]:
def get_accuracy(model, data, batch_size=128):
    """Return accuracy over `data`.
    Use `num_workers=0` to avoid worker-based nondeterminism and a fixed batch size.
    """
    model.eval()

    # Using num_workers=0 avoids separate worker RNG states which can affect reproducibility
    loader = DataLoader(data, batch_size=batch_size, shuffle=False, num_workers=0)
    correct, total = 0, 0

    with torch.no_grad():
        for imgs, labels in loader:
            imgs, labels = imgs.to(device), labels.to(device)

            out = model(imgs)                # logits: (B, 10)
            preds = out.argmax(dim=1)        # predicted class index
            correct += (preds == labels).sum().item()
            total += labels.size(0)

    return correct / total

4.2 Average loss function (for validation loss curve)¶

This computes the mean cross-entropy loss over an entire dataset. We need this to plot the validation loss each epoch.

In [12]:
def get_avg_loss(model, data, criterion, batch_size=128):
    """Compute mean cross-entropy loss over `data`.
    Uses `num_workers=0` for reproducibility.
    """
    model.eval()

    loader = DataLoader(data, batch_size=batch_size, shuffle=False, num_workers=0)
    total_loss, total_n = 0.0, 0

    with torch.no_grad():
        for imgs, labels in loader:
            imgs, labels = imgs.to(device), labels.to(device)

            out = model(imgs)
            loss = criterion(out, labels)

            # multiply by batch size so we can average properly over the whole dataset
            total_loss += loss.item() * labels.size(0)
            total_n += labels.size(0)

    return total_loss / total_n

4.3 Training function¶

This follows the notebook pattern: train(...) accepts the dataset/list, and creates its own DataLoader. I also package in:

  • compute validation loss each epoch (for the required plot)
  • track train/val accuracy (useful for reporting validation error)

def train(model, train_data, val_data, batch_size=128, num_epochs=20, lr=1e-3, weight_decay=0.0):
    """Train `model` and return lists of train/val losses and accuracies.

    Determinism notes:
    - We pass a seeded `g` generator (created earlier) into DataLoader so shuffle order is repeatable.
    - `num_workers=0` avoids per-worker RNG nondeterminism.
    - Recreate / initialize the model after setting seeds to ensure consistent weight init.
    """
    criterion = nn.CrossEntropyLoss()
    optimizer = optim.AdamW(model.parameters(), lr=lr, weight_decay=weight_decay)

    # Use seeded generator `g` and single-process workers for determinism
    # (DataLoader will shuffle using the provided generator)
    train_loader = DataLoader(train_data, batch_size=batch_size, shuffle=True, generator=g, num_workers=0)

    # Store evaluation metrics
    train_losses, val_losses = [], []
    train_accs, val_accs = [], []

    for epoch in range(num_epochs):
        model.train()
        running_loss = 0.0
        # Iterate over mini-batches
        for imgs, labels in train_loader:
            imgs, labels = imgs.to(device), labels.to(device)
            # Forward pass 
            out = model(imgs)
            loss = criterion(out, labels)
            # Backward pass and optimization step
            optimizer.zero_grad()
            loss.backward()
            nn.utils.clip_grad_norm_(model.parameters(), 1.0)  # possibly rescales p.grad
            optimizer.step()

            running_loss += loss.item() * labels.size(0)

        # --- end of epoch: compute metrics ---
        avg_train_loss = running_loss / len(train_data)
        avg_val_loss = get_avg_loss(model, val_data, criterion, batch_size=batch_size)

        train_acc = get_accuracy(model, train_data, batch_size=batch_size)
        val_acc = get_accuracy(model, val_data, batch_size=batch_size)

        train_losses.append(avg_train_loss)
        val_losses.append(avg_val_loss)
        train_accs.append(train_acc)
        val_accs.append(val_acc)

        print(f"Epoch {epoch+1}/{num_epochs} | "
              f"TrainLoss {avg_train_loss:.4f} | ValLoss {avg_val_loss:.4f} | "
              f"TrainAcc {train_acc:.4f} | ValAcc {val_acc:.4f}")

    return train_losses, val_losses, train_accs, val_accs

def summarize_run(name, train_losses, val_losses, train_accs, val_accs):
    final_train_acc = train_accs[-1]
    final_val_acc = val_accs[-1]
    final_val_error = 1 - final_val_acc

    peak_val_acc = max(val_accs)
    peak_epoch = val_accs.index(peak_val_acc) + 1

    gap = final_train_acc - final_val_acc

    print(f"\n{name}")
    print(f"  Final val acc: {final_val_acc:.4f}  | Final val error: {final_val_error:.4f}")
    print(f"  Peak  val acc: {peak_val_acc:.4f}  (epoch {peak_epoch})")
    print(f"  Overfit gap:   {gap:.4f}  (final TrainAcc - ValAcc)")

7.3 Variant: Dropout (p = 0.2)¶

Motivation. The baseline MLP showed a generalisation gap (TrainAcc > ValAcc), suggesting overfitting. Dropout is a standard MLP regulariser that reduces co-adaptation by randomly zeroing hidden activations during training.

How to report this run accurately. Use the printed summarize_run(...) output and the plotted curves from this section to record:

  • final validation accuracy / validation error
  • peak validation accuracy and epoch
  • final overfitting gap (TrainAcc − ValAcc)

Interpretation guide.

  • If the train/validation gap shrinks relative to baseline, dropout is helping regularisation.
  • If peak validation accuracy occurs well before the final epoch, use that to motivate early stopping or additional tuning.
  • If validation loss is noisy, test weight decay and/or learning-rate tuning next.

7.13 Add Mild Data Augmentation + Label Smoothing (Final Improvement)¶

Motivation. The loss curves show classic overfitting: training loss continues to fall while validation loss reaches a minimum early and then rises.
We apply two standard generalisation techniques without changing the model family:

  2. Random padded crops to encourage robustness to small translations.

Implementation note. Augmentations must be applied on-the-fly via dataset transforms, so we use torch.utils.data.Subset instead of converting the dataset to a Python list.

import torch.nn as nn
import torch.optim as optim
from torch.utils.data import DataLoader

def train(
    model, train_data, val_data,
    batch_size=128, num_epochs=20, lr=0.0001, weight_decay=0.0001,
    label_smoothing=0.02, # small amount of label smoothing for regularization
    eval_every=5,        # compute accuracies every N epochs
    val_every=1,         # compute val loss every N epochs (use 1 for final report)
    progress_batches=50  # print batch progress on epoch 1 every N batches (None to disable)
):
    criterion = nn.CrossEntropyLoss(label_smoothing=label_smoothing)
    optimizer = optim.AdamW(model.parameters(), lr=lr, weight_decay=weight_decay)

    # Keep this Mac/MPS-friendly + deterministic
    train_loader = DataLoader(train_data, batch_size=batch_size, shuffle=True, generator=g, num_workers=0)

    train_losses, val_losses = [], []
    train_accs, val_accs = [], []

    for epoch in range(num_epochs):
        model.train()
        running_loss = 0.0

        for b, (imgs, labels) in enumerate(train_loader, start=1):
            imgs, labels = imgs.to(device), labels.to(device)

            out = model(imgs)
            loss = criterion(out, labels)

            optimizer.zero_grad()
            loss.backward()
            optimizer.step()

            running_loss += loss.item() * labels.size(0)

            # visibility so it never feels "stuck"
            if (epoch == 0) and (progress_batches is not None) and (b % progress_batches == 0):
                print(f"  [epoch 1] batch {b} / ~{len(train_loader)} | batch loss {loss.item():.4f}")

        avg_train_loss = running_loss / len(train_data)

        # val loss (optionally less frequent while experimenting)
        if (epoch + 1) % val_every == 0:
            avg_val_loss = get_avg_loss(model, val_data, criterion, batch_size=batch_size)
        else:
            avg_val_loss = float("nan")

        # accuracies (expensive; do less often)
        if (epoch + 1) % eval_every == 0 or epoch == 0 or epoch == num_epochs - 1:
            train_acc = get_accuracy(model, train_data, batch_size=batch_size)
            val_acc   = get_accuracy(model, val_data, batch_size=batch_size)
        else:
            train_acc = float("nan")
            val_acc   = float("nan")

        train_losses.append(avg_train_loss)
        val_losses.append(avg_val_loss)
        train_accs.append(train_acc)
        val_accs.append(val_acc)

        # nice ST311-style print
        msg = f"Epoch {epoch+1}/{num_epochs} | TrainLoss {avg_train_loss:.4f}"
        if not (avg_val_loss != avg_val_loss):  # not NaN
            msg += f" | ValLoss {avg_val_loss:.4f}"
        if not (train_acc != train_acc):        # not NaN
            msg += f" | TrainAcc {train_acc:.4f} | ValAcc {val_acc:.4f}"
        print(msg)

    return train_losses, val_losses, train_accs, val_accs
